{{/* Render hook for Markdown links. Add rel security attrs for external links; keep internal links clean. */}} {{- $u := .Destination | safeURL -}} {{- $isAbs := or (hasPrefix $u "http://") (hasPrefix $u "https://") -}} {{- if $isAbs -}} {{ .Text | safeHTML }} {{- else -}} {{ .Text | safeHTML }} {{- end -}}
