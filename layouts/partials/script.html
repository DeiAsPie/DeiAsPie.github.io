Usage: {{ partial "script" (dict "path" "js/init-ui.js" "defer" true) }} */ -}} {{- $path := .path -}} {{- if not $path -}} {{- errorf "partial 'script' requires a 'path' parameter" -}} {{- end -}} {{- $defer := default false .defer -}} {{- $res := resources.Get $path -}} {{- if not hugo.IsServer -}} {{- $res = $res | minify | fingerprint -}} {{- end -}} {{- if $defer -}} {{- else -}} {{- end -}}
